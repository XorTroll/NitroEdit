267 inline constexpr Result GetResultDescription(const Result rc, std::string &out_desc) {
268 for(u32 i = 0; i < std::size(ResultDescriptionTable); i++) {
269 if(ResultDescriptionTable[i].first.value == rc.value) {
270 out_desc = ResultDescriptionTable[i].second;
271 NTR_R_SUCCEED();
272 }
273 }

276 }
